01 Mocking a service with side effects

The orderService below charges a payment, then sends a confirmation email. Both are external calls — your tests must never fire either one for real. Your job: mock both dependencies and verify the business logic in between.

// paymentService.js 
async function charge(userId, amount) {
  // calls Stripe — never run this in tests
  return { success: true, transactionId: 'txn_live_abc123' };
}
module.exports = { charge };
// emailService.js 
async function sendOrderConfirmation(email, transactionId) {
  // calls SendGrid — never run this in tests
  return { sent: true };
}
module.exports = { sendOrderConfirmation };
// orderService.js — this is what you're testing 
const { charge }                  = require('./paymentService');
const { sendOrderConfirmation }   = require('./emailService');

async function placeOrder(userId, email, amount) {
  if (amount <= 0) throw new Error('Invalid amount');

  const result = await charge(userId, amount);
  if (!result.success) throw new Error('Payment failed');

  await sendOrderConfirmation(email, result.transactionId);
  return { orderId: `order_${Date.now()}`, transactionId: result.transactionId };
}

module.exports = { placeOrder };
🧩 Your task — write orderService.test.js
  • Mock both paymentService and emailService modules entirely
  • Test: a valid order returns an object with both orderId and transactionId
  • Test: sendOrderConfirmation is called with the correct email and the transactionId that came back from charge
  • Test: an amount of 0 throws 'Invalid amount' — and charge is never called
  • Test: when charge resolves with { success: false }, the function throws 'Payment failed' — and sendOrderConfirmation is never called bonus
💡 Hint
Use expect(fn).not.toHaveBeenCalled() to assert a mock was never touched. Reset mocks between tests with beforeEach(() => jest.clearAllMocks()).
⚠️ Watch out
jest.mock() hoists to the top of the file — declare it before any require calls. Don't try to conditionally mock inside a test.
02 Async — retry logic

fetchWithRetry calls an external API and retries up to maxRetries times on failure. Testing retry logic is tricky — you need the same mock to behave differently on consecutive calls. This exercise is about sequencing mock responses.

// apiClient.js 
const axios = require('axios');

async function getData(url) {
  const { data } = await axios.get(url);
  return data;
}

module.exports = { getData };
// fetchWithRetry.js — this is what you're testing 
const { getData } = require('./apiClient');

async function fetchWithRetry(url, maxRetries = 3) {
  let lastError;

  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      return await getData(url);
    } catch (err) {
      lastError = err;
    }
  }

  throw new Error(`Failed after ${maxRetries} attempts: ${lastError.message}`);
}

module.exports = { fetchWithRetry };
🧩 Your task — write fetchWithRetry.test.js
  • Mock apiClient. Test: first attempt succeeds — getData is called exactly once and the data is returned
  • Test: first attempt fails, second succeeds — getData is called exactly twice and the data is still returned correctly
  • Test: all 3 attempts fail — the function throws 'Failed after 3 attempts: …' and getData was called 3 times
  • Test: passing maxRetries = 1 with a failing mock throws after exactly 1 call bonus
💡 Hint — sequencing responses
Chain mockResolvedValueOnce / mockRejectedValueOnce calls — each one is consumed in order: 
getData
  .mockRejectedValueOnce(new Error('timeout'))   // attempt 1 fails
  .mockResolvedValueOnce({ id: 1 });              // attempt 2 succeeds
03 MongoDB with mongodb-memory-server

You have a Product model and a productService that queries it directly. Wire up an in-memory MongoDB instance and test the full Mongoose query path — including schema validation and the duplicate-slug guard.

📦 Dependencies needed
npm install --save-dev mongodb-memory-server mongoose
// models/Product.js 
const mongoose = require('mongoose');

const productSchema = new mongoose.Schema({
  name:     { type: String, required: true },
  slug:     { type: String, required: true, unique: true },
  price:    { type: Number, required: true, min: 0 },
  inStock:  { type: Boolean, default: true },
});

module.exports = mongoose.model('Product', productSchema);
// services/productService.js 
const Product = require('../models/Product');

async function createProduct(data) {
  const existing = await Product.findOne({ slug: data.slug });
  if (existing) throw new Error('Slug already in use');
  return Product.create(data);
}

async function getAvailableProducts() {
  return Product.find({ inStock: true }).lean();
}

async function discontinue(slug) {
  const product = await Product.findOneAndUpdate(
    { slug },
    { inStock: false },
    { new: true }
  );
  if (!product) throw new Error('Product not found');
  return product;
}

module.exports = { createProduct, getAvailableProducts, discontinue };
🧩 Your task — write productService.test.js

Start with the memory-server lifecycle hooks, then write the tests below.

  • Set up beforeAll / afterEach / afterAll correctly using MongoMemoryServer
  • createProduct — happy path: product is created and returned with the correct name, slug, and inStock: true default
  • createProduct — duplicate slug: throws 'Slug already in use'
  • createProduct — schema validation: a negative price is rejected by Mongoose (min: 0)
  • getAvailableProducts — returns only products where inStock is true; products where inStock: false are excluded
  • discontinue — sets inStock to false and returns the updated product bonus
  • discontinue — throws 'Product not found' for an unknown slug bonus
💡 Hint — test isolation
Each test should await Product.deleteMany({}) in afterEach — not afterAll. This keeps tests independent without restarting the server between every test.
⚠️ Schema validation errors
Mongoose throws a ValidationError, not a plain Error. Match on the message text with .rejects.toThrow('min') or check err.name === 'ValidationError' in a try/catch.
💡 Async rendering
The component updates after the promise resolves — always wrap post-fetch assertions in await waitFor(...) or use await screen.findByText(...).
